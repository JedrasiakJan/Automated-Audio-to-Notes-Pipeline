events {
    worker_connections 1024;
}

http {
    client_max_body_size 500M;

    # 1. PRZEKIEROWANIE Z HTTP NA HTTPS
    server {
        listen 80;
        server_name twoja-domena.pl www.twoja-domena.pl; # Wpisz tutaj swoją domenę
        return 301 https://$host$request_uri;
    }

    # 2. GŁÓWNA KONFIGURACJA HTTPS
    server {
        listen 443 ssl;
        server_name twoja-domena.pl www.twoja-domena.pl;

        # Certyfikaty MUSZĄ istnieć na serwerze pod spodem!
        ssl_certificate /etc/nginx/ssl/fullchain.pem;
        ssl_certificate_key /etc/nginx/ssl/privkey.pem;

        # Ochrona kryptograficzna
        ssl_protocols TLSv1.2 TLSv1.3;
        ssl_ciphers HIGH:!aNULL:!MD5;
        ssl_prefer_server_ciphers on;

        location / {
            proxy_pass http://medical-ai:8501;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            
            # WebSocket Support
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection "upgrade";
            
            # Timeouts
            proxy_read_timeout 86400;
            proxy_send_timeout 86400;
            proxy_connect_timeout 86400;

            # Autentykacja
            auth_basic "Dostep zastrzezony dla personelu";
            auth_basic_user_file /etc/nginx/.htpasswd;
        }
    }
}
